Korisinici :


CREATE TABLE [dbo].[Korisnici] (
    [Id_Korisnik]    INT           IDENTITY (1, 1) NOT NULL,
    [Lozinka]        NVARCHAR (50) NOT NULL,
    [Ime]            NVARCHAR (50) NOT NULL,
    [Prezime ]       NVARCHAR (50) NOT NULL,
    [Email]          NVARCHAR (50) NOT NULL,
    [Datum_rodjenja] DATETIME      NOT NULL,
    [Broj_telefona]  NVARCHAR (50) NULL,
    [Stanodavac]     NVARCHAR (50) NULL,
    PRIMARY KEY CLUSTERED ([Id_Korisnik] ASC)
);


Stanovi :

CREATE TABLE [dbo].[Stanovi] (
    [Id_Stan]        INT           IDENTITY (1, 1) NOT NULL,
    [Mesto]          NVARCHAR (50) NOT NULL,
    [Adresa]         NVARCHAR (50) NOT NULL,
    [Broj_soba]      INT           NOT NULL,
    [Kvadratura]     INT           NOT NULL,
    [Postanski_broj] INT           NOT NULL,
    [Cena_nocenja]   FLOAT (53)    NOT NULL,
    [Depozit]        FLOAT (53)    NOT NULL,
    [Broj_sprata]    INT           NULL,
    PRIMARY KEY CLUSTERED ([Id_Stan] ASC),
    CONSTRAINT [FK_id_korisnika] FOREIGN KEY ([Id_Stan]) REFERENCES [dbo].[Korisnici] ([Id_Korisnik])
);






Rente :

CREATE TABLE [dbo].[Rente] (
    [Id] INT NOT NULL IDENTITY,
    [Datum_pocetka] DATETIME NOT NULL, 
    [Broj_dana] INT NOT NULL, 
    [Nacin_placanja] NVARCHAR(50) NOT NULL, 
    [Broj_kartice] NVARCHAR(50) NULL, 
    PRIMARY KEY CLUSTERED ([Id] ASC), 
    CONSTRAINT [FK_Rente_Korisnici] FOREIGN KEY ([Id]) REFERENCES [Korisnici]([Id_Korisnik]), 
    CONSTRAINT [FK_Rente_Stanovi] FOREIGN KEY ([Id]) REFERENCES [Stanovi]([Id_Stan]), 
    CONSTRAINT [FK_Rente_Stanovi_Korisnici] FOREIGN KEY ([Id]) REFERENCES [Stanovi]([FK_id_korisnika]), 
    

);
